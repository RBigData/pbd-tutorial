\subsection{Example}

\begin{frame}[fragile]
  \begin{block}{Example Script}
\begin{lstlisting}[title=my\_svd.r]
library(pbdMPI, lib.loc="~/R/prof", quietly=TRUE)
library(pbdDMAT, quietly=T)
init.grid()

n <- 1000
x <- ddmatrix("rnorm", n, n)

asdf <- La.svd(x)


finalize()
\end{lstlisting}
  \end{block}
\end{frame}
